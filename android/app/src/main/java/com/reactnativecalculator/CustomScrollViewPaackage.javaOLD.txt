package com.reactnativecalculator;

import com.facebook.react.ReactPackage;
import com.facebook.react.bridge.NativeModule;
import com.facebook.react.bridge.ReactApplicationContext;
import com.facebook.react.uimanager.ViewManager;

import java.util.Arrays;
import java.util.Collections;
import java.util.List;

import com.reactnativecalculator.MyCustomScrollViewManager;

public class CustomScrollViewPackage implements ReactPackage {

    @Override
    public List<NativeModule> createNativeModules(ReactApplicationContext reactContext) {
        return Collections.emptyList();
    };

    @Override
    public List<ViewManager> createViewManagers(ReactApplicationContext reactContext) {
        // Register your custom view manager here
        return Arrays.<ViewManager>asList( MyCustomScrollViewManager() )
        //new CustomScrollViewManager()
    }
}
